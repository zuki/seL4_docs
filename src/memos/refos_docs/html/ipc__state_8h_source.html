25 refosio_internal_save_IPC_buffer(void)
26 {

28  assert(_refosioIPCBufferBackupStack >= 0);
29  seL4_IPCBuffer *ipcBuf = &_refosioIPCBufferBackup[_refosioIPCBufferBackupStack++];

31 #if defined(ARCH_ARM)
32  memcpy(ipcBuf, seL4_GetIPCBuffer(), sizeof(seL4_IPCBuffer));
33 #elif defined(ARCH_IA32)
34  seL4_Word *dest = (seL4_Word *) ipcBuf;

36  #ifdef CONFIG_X86_64
37  asm volatile ("movq %%gs:0(,%1,0x8), %0" : "=r"(*(dest + i)) : "r"(i) : "memory");
38  #else

40  #endif
41  }
42 #else
43  #error "Unknown arch."
44 #endif
45 }
