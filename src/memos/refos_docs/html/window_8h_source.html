181 w_convert_permission_to_caprights(seL4_Word permission)
182 {
183  seL4_CapRights capr = seL4_CanGrant;
184  if (permission | W_PERMISSION_WRITE) {
185  capr |= seL4_CanWrite;
186  }
187  if (permission | W_PERMISSION_READ) {
188  capr |= seL4_CanRead;
189  }
190  return capr;
